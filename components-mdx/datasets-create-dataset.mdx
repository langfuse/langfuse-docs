import { LangfuseClient } from "@langfuse/client"

const langfuse = new LangfuseClient()

  // optional description
  description: "My first dataset",
  // optional metadata
  metadata: {
    author: "Alice",
    date: "2022-01-01",
    type: "benchmark",
  },
});

import { Home } from "../components/home";
import { useData } from "nextra/data";

export function HomePage() {
  const { releases } = useData();
  return <Home releases={releases} />;
}

export const getStaticProps = async () => {
  const headers = {
    Accept: "application/vnd.github.v3+json",
    ...(process.env.GITHUB_ACCESS_TOKEN
      ? { Authorization: `token ${process.env.GITHUB_ACCESS_TOKEN}` }
      : {}),
  };
  const res = await fetch(
    "https://api.github.com/repos/langfuse/langfuse/releases",
    { headers }
  );
  const resJs = await fetch(
    "https://api.github.com/repos/langfuse/langfuse-js/releases",
    { headers }
  );
  const resPython = await fetch(
    "https://api.github.com/repos/langfuse/langfuse-python/releases",
    { headers }
  );
  const releasesLangfuse = await res.json();
  const releasesLangfuseJs = await resJs.json();
  const releasesLangfusePython = await resPython.json();
  if (!res.ok || !resJs.ok || !resPython.ok) {
    return {
      props: {
        ssg: {
          releases: [],
        },
      },
    };
  }
  return {
    props: {
      ssg: {
        releases: {
          langfuse: releasesLangfuse
            .filter(
              (release) =>
                !release.draft &&
                !release.pre_release &&
                new Date(release.created_at).getTime() >
                  new Date().getTime() - 14 * 24 * 60 * 60 * 1000 // 14 days
            )
            .map((release) => ({
              name: release.name,
              html_url: release.html_url,
              created_at: release.created_at,
              repo: "langfuse/langfuse",
            }))
            .sort(
              (a, b) =>
                new Date(b.created_at).getTime() -
                new Date(a.created_at).getTime()
            ),
          jsSdk: releasesLangfuseJs
            .filter(
              (release) =>
                !release.draft &&
                !release.pre_release &&
                new Date(release.created_at).getTime() >
                  new Date().getTime() - 14 * 24 * 60 * 60 * 1000 // 14 days
            )
            .map((release) => ({
              name: release.name,
              html_url: release.html_url,
              created_at: release.created_at,
              repo: "langfuse/langfuse-js",
            }))
            .sort(
              (a, b) =>
                new Date(b.created_at).getTime() -
                new Date(a.created_at).getTime()
            ),
          pythonSdk: releasesLangfusePython
            .filter(
              (release) =>
                !release.draft &&
                !release.pre_release &&
                new Date(release.created_at).getTime() >
                  new Date().getTime() - 14 * 24 * 60 * 60 * 1000 // 14 days
            )
            .map((release) => ({
              name: release.name,
              html_url: release.html_url,
              created_at: release.created_at,
              repo: "langfuse/langfuse-python",
            }))
            .sort(
              (a, b) =>
                new Date(b.created_at).getTime() -
                new Date(a.created_at).getTime()
            ),
        },
      },
    },
  };
};

<HomePage />
